function readFile(e) {            
     var file = e.target.files[0];
     var reader = new FileReader();
     reader.onload = function() {        
        var lines = this.result.split('\n');
        for (var line = 0; line < lines.length; line++) {
            var strings = lines[line].trimRight().split(' ');                
            switch (strings[0]) {
                case ('v'):                    
                    break;            
                case('f'):
                    break;
            }                
         }      
     };
     reader.readAsText(file);
}
